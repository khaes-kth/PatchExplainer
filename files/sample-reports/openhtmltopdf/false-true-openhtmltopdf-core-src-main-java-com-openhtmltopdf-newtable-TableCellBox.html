         408:            // The following rules apply for resolving conflicts and figuring out which
         409:            // border
         410:            // to use.
         411:            // (1) Borders with the 'border-style' of 'hidden' take precedence over all
         412:            // other conflicting
         413:            // borders. Any border with this value suppresses all borders at this
         414:            // location.
         415:            // (2) Borders with a style of 'none' have the lowest priority. Only if the
         416:            // border properties of all
         417:            // the elements meeting at this edge are 'none' will the border be omitted
         418:            // (but note that 'none' is
         419:            // the default value for the border style.)
         420:            // (3) If none of the styles are 'hidden' and at least one of them is not
         421:            // 'none', then narrow borders
         422:            // are discarded in favor of wider ones. If several have the same
         423:            // 'border-width' then styles are preferred
         424:            // in this order: 'double', 'solid', 'dashed', 'dotted', 'ridge', 'outset',
         425:            // 'groove', and the lowest: 'inset'.
         426:            // (4) If border styles differ only in color, then a style set on a cell
         427:            // wins over one on a row,
         428:            // which wins over a row group, column, column group and, lastly, table. It
         429:            // is undefined which color
         430:            // is used when two elements of the same type disagree.
         431:            public static CollapsedBorderValue compareBorders(
         432:                    CollapsedBorderValue border1, CollapsedBorderValue border2, boolean returnNullOnEqual) {
         433:                // Sanity check the values passed in.  If either is null, return the other.
         434:                if (!border2.defined()) {
         435:                    return border1;
         436:                }
         437:                
         438:                if (!border1.defined()) {
         439:                    return border2;
         440:                }
         441:                
         442:                // Rule #1 above.
         443:                if (border1.style() == IdentValue.HIDDEN)
         444:                {
         445:                    return border1;
         446:                }
         447:                if (border2.style() == IdentValue.HIDDEN)
         448:                {
         449:                    return border2;
         450:                }
         451:        
         452:                // Rule #2 above. A style of 'none' has lowest priority and always loses
         453:                // to any other border.
         454:                if (border2.style() == IdentValue.NONE) {
         455:                    return border1;
         456:                }
         457:        
         458:                if (border1.style() == IdentValue.NONE) {
         459:                    return border2;
         460:                }
         461:        
         462:                // The first part of rule #3 above. Wider borders win.
         463:                if (border1.width() != border2.width()) {
         464:                    return border1.width() > border2.width() ? border1 : border2;
         465:                }
         466:        
         467:                // The borders have equal width. Sort by border style.
         468:                if (border1.style() != border2.style()) {
         469:                    return BORDER_PRIORITIES[border1.style().FS_ID] > 
         470:                        BORDER_PRIORITIES[border2.style().FS_ID] ? border1 : border2;
         471:                }

         473:                // The border have the same width and style. Rely on precedence (cell
         474:                // over row over row group, etc.)
         475:                if (returnNullOnEqual && border1.precedence() == border2.precedence()) {
         476:                    return null;
         477:                } else {
         478:                    return border1.precedence() >= border2.precedence() ? border1 : border2;    
         479:                }

      413->N:            // The following rules apply for resolving conflicts and figuring out which
      414->N:            // border
      415->N:            // to use.
      416->N:            // (1) Borders with the 'border-style' of 'hidden' take precedence over all
      417->N:            // other conflicting
      418->N:            // borders. Any border with this value suppresses all borders at this
      419->N:            // location.
      420->N:            // (2) Borders with a style of 'none' have the lowest priority. Only if the
      421->N:            // border properties of all
      422->N:            // the elements meeting at this edge are 'none' will the border be omitted
      423->N:            // (but note that 'none' is
      424->N:            // the default value for the border style.)
      425->N:            // (3) If none of the styles are 'hidden' and at least one of them is not
      426->N:            // 'none', then narrow borders
      427->N:            // are discarded in favor of wider ones. If several have the same
      428->N:            // 'border-width' then styles are preferred
      429->N:            // in this order: 'double', 'solid', 'dashed', 'dotted', 'ridge', 'outset',
      430->N:            // 'groove', and the lowest: 'inset'.
      431->N:            // (4) If border styles differ only in color, then a style set on a cell
      432->N:            // wins over one on a row,
      433->N:            // which wins over a row group, column, column group and, lastly, table. It
      434->N:            // is undefined which color
      435->N:            // is used when two elements of the same type disagree.

    440->435:                    return border1;
      441->N:                }
      442->N:                
    443->438:                if (!border1.defined()) {
    444->439:                    return border2;
      445->N:                }
      446->N:                
      447->N:                // Rule #1 above.
    448->443:                if (border1.style() == IdentValue.HIDDEN)
      449->N:                {
    450->445:                    return border1;
      451->N:                }
    452->447:                if (border2.style() == IdentValue.HIDDEN)
      453->N:                {
    454->449:                    return border2;
      455->N:                }
      456->N:        
      457->N:                // Rule #2 above. A style of 'none' has lowest priority and always loses
      458->N:                // to any other border.

    460->455:                    return border1;
