      675->N:             * Checks if this line box crosses a page break and if so moves it to
      676->N:             * the next page.
      677->N:             * Also takes care that in-flow lines do not overlap footnote content.
      678->N:             */
      679->U:            public void checkPagePosition(LayoutContext c, boolean alwaysBreak) {
      680->U:                if (! c.isPageBreaksAllowed()) {
      681->U:                    return;
      682->U:                }
      683->U:        
      684->U:                PageBox pageBox = c.getRootLayer().getFirstPage(c, this);
      685->U:        
      686->U:                if (pageBox != null) {
      687->U:                    // We need to force a page break if any of our content goes over a page break,
      688->U:                    // otherwise we will get repeated content in page margins (because content is
      689->U:                    // printed on both pages).
      690->U:        
      691->U:                    // Painting top and bottom take account of line-height other than 1.
      692->U:                    int greatestAbsY = getMaxPaintingBottom();
      693->U:                    int leastAbsY = getMinPaintingTop();
      694->U:        
      695->U:                    boolean overflowsPage;
      696->U:                    if (c.isInFloatBottom()) {
      697->U:                        // For now we don't support paginated tables in float:bottom content.
      698->U:                        overflowsPage = greatestAbsY >= pageBox.getBottom();
      699->U:                    } else {
      700->U:                        overflowsPage = greatestAbsY >= pageBox.getBottom(c) - c.getExtraSpaceBottom();
      701->U:                    }
      702->U:        
      703->U:                    boolean tooBig = (greatestAbsY - leastAbsY) > pageBox.getContentHeight(c);
      704->U:                    boolean needsPageBreak = alwaysBreak || (overflowsPage && !tooBig); 
      705->U:        
      706->U:                   if (needsPageBreak) {
      707->U:                       beforeChangePage(c);
      708->U:        
      709->U:                       forcePageBreakBefore(c, IdentValue.ALWAYS, false, leastAbsY);
      710->U:                       calcCanvasLocation();
      711->U:        
      712->U:                       checkFootnoteReservedPage(c, c.getRootLayer().getFirstPage(c, this), false);
      713->U:        
      714->U:                       afterChangePage(c);
      715->U:                   } else if (pageBox.getTop() + c.getExtraSpaceTop() > getAbsY()) {
      716->U:                       // It is in the extra room at the top!
      717->U:                       int diff = pageBox.getTop() + c.getExtraSpaceTop() - getAbsY();
      718->U:                       setY(getY() + diff);
      719->U:                       calcCanvasLocation();
      720->U:        
      721->U:                       checkFootnoteReservedPage(c, pageBox, true);
      722->U:                   } else {
      723->U:                       checkFootnoteReservedPage(c, pageBox, true);
      724->U:                   }
      725->U:                }
      726->U:            }

      768->U:        
      769->U:                       if (runHooks) {
